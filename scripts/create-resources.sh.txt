#!/bin/bash
# Script fictício para criar recursos AWS

echo "Criando bucket S3..."
aws s3 mb s3://abstergo-bucket

echo "Criando tabela DynamoDB..."
aws dynamodb create-table \
    --table-name Catalogo \
    --attribute-definitions AttributeName=ProductId,AttributeType=S \
    --key-schema AttributeName=ProductId,KeyType=HASH \
    --billing-mode PAY_PER_REQUEST

echo "Criando função Lambda..."
aws lambda create-function \
    --function-name processOrder \
    --runtime python3.11 \
    --handler lambda_function.lambda_handler \
    --role arn:aws:iam::123456789012:role/exemplo-lambda-role \
    --zip-file fileb://function.zip

echo "Provisionamento fictício concluído!"
